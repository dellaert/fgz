\def\fgz@label@sep{0pt}

\newcommand{\fgz@applystyles}{%
  \tikzset{
    fgz var/.style={draw=fgz@stroke, fill=white, circle, minimum size=\fgz@node@size, inner sep=\fgz@label@sep, line width=0.9pt, font=\fgz@label@font},
    fgz known/.style={draw=fgz@stroke, fill=white, rectangle, minimum size=\fgz@node@size, inner sep=\fgz@label@sep, line width=0.9pt, font=\fgz@label@font},
    fgz factor/.style={draw=fgz@factor, fill=fgz@factor, circle, minimum size=\fgz@factor@size, inner sep=0pt},
    fgz factor square/.style={draw=fgz@factor, fill=fgz@factor, rectangle, minimum size=\fgz@factor@size, inner sep=0pt},
    fgz edge/.style={draw=fgz@stroke, line width=0.9pt},
    fgz edge directed/.style={fgz edge, -{Stealth[length=2.2mm,width=1.8mm]}}
  }%
}

\newcommand{\fgzsetnodesize}[1]{\def\fgz@node@size{#1}\fgz@applystyles}
\newcommand{\fgzsetfactorsize}[1]{\def\fgz@factor@size{#1}\fgz@applystyles}

\newcommand{\fgzsettheme}[1]{%
  \ifcsname fgz@theme@#1\endcsname
    \csname fgz@theme@#1\endcsname
  \else
    \fgz@theme@classic
  \fi
  \fgz@applystyles
}
