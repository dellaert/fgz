\newcommand{\fgz@applystyles}{%
  \tikzset{
    fgz var/.style={draw=fgz@var@draw, fill=fgz@var@fill, circle, minimum size=\fgz@node@size, inner sep=\fgz@label@sep, line width=\fgz@node@line@width, font=\fgz@label@font},
    fgz known/.style={draw=fgz@known@draw, fill=fgz@known@fill, rectangle, minimum size=\fgz@node@size, inner sep=\fgz@label@sep, line width=\fgz@node@line@width, font=\fgz@label@font},
    fgz factor/.style={draw=fgz@factor@draw, fill=fgz@factor@fill, circle, minimum size=\fgz@factor@size, inner sep=0pt},
    fgz factor square/.style={draw=fgz@factor@draw, fill=fgz@factor@fill, rectangle, minimum size=\fgz@factor@size, inner sep=0pt},
    fgz edge/.style={draw=fgz@stroke, line width=\fgz@edge@line@width},
    fgz edge directed/.style={fgz edge, -{Stealth[length=2.2mm,width=1.8mm]}}
  }%
}
