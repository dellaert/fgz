\usetikzlibrary{arrows.meta}

\definecolor{fgzClassicStroke}{HTML}{1F2933}
\definecolor{fgzClassicFactor}{HTML}{111827}
\definecolor{fgzTextbookStroke}{HTML}{0F172A}
\definecolor{fgzTextbookFactor}{HTML}{334155}
\definecolor{fgzBlogStroke}{HTML}{7C2D12}
\definecolor{fgzBlogFactor}{HTML}{C2410C}

\colorlet{fgz@stroke}{fgzClassicStroke}
\colorlet{fgz@factor}{fgzClassicFactor}

\newcommand{\fgz@applystyles}{%
  \tikzset{
    fgz var/.style={draw=fgz@stroke, circle, minimum size=8mm, inner sep=0pt, line width=0.9pt},
    fgz known/.style={draw=fgz@stroke, rectangle, minimum size=8mm, inner sep=0pt, line width=0.9pt},
    fgz factor/.style={draw=fgz@factor, fill=fgz@factor, circle, minimum size=3.2mm, inner sep=0pt},
    fgz factor square/.style={draw=fgz@factor, fill=fgz@factor, rectangle, minimum size=3.2mm, inner sep=0pt},
    fgz edge/.style={draw=fgz@stroke, line width=0.9pt},
    fgz edge directed/.style={fgz edge, -{Stealth[length=2.2mm,width=1.8mm]}}
  }%
}

\newcommand{\fgzsettheme}[1]{%
  \ifcsname fgz@theme@#1\endcsname
    \csname fgz@theme@#1\endcsname
  \else
    \fgz@theme@classic
  \fi
  \fgz@applystyles
}

\newcommand{\fgz@theme@classic}{%
  \colorlet{fgz@stroke}{fgzClassicStroke}%
  \colorlet{fgz@factor}{fgzClassicFactor}%
}

\newcommand{\fgz@theme@textbook}{%
  \colorlet{fgz@stroke}{fgzTextbookStroke}%
  \colorlet{fgz@factor}{fgzTextbookFactor}%
}

\newcommand{\fgz@theme@blog}{%
  \colorlet{fgz@stroke}{fgzBlogStroke}%
  \colorlet{fgz@factor}{fgzBlogFactor}%
}

\newcommand{\fgzVar}[4]{\node[fgz var] (#1) at (#2,#3) {#4};}
\newcommand{\fgzKnown}[4]{\node[fgz known] (#1) at (#2,#3) {#4};}
\newcommand{\fgzFactor}[3]{\node[fgz factor] (#1) at (#2,#3) {};}
\newcommand{\fgzFactorSquare}[3]{\node[fgz factor square] (#1) at (#2,#3) {};}
\newcommand{\fgzEdgeU}[2]{\draw[fgz edge] (#1) -- (#2);}
\newcommand{\fgzEdgeD}[2]{\draw[fgz edge directed] (#1) -- (#2);}
\newcommand{\fgzCurveU}[4]{\draw[fgz edge] (#1) .. controls (#3,#4) and (#3,#4) .. (#2);}
\newcommand{\fgzCurveD}[4]{\draw[fgz edge directed] (#1) .. controls (#3,#4) and (#3,#4) .. (#2);}
